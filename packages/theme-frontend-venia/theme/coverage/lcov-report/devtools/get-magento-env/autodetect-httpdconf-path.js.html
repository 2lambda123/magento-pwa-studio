const ECRunner = require('../external-command-runner');
const psl = require('ps-list');
const KNOWN_HTTPDS = ['httpd', 'httpd24', 'apache', 'apache2'];
// warning text gets used in two places, make a convenience function
const warnFor = (logger, cmd, result) => {
    logger.warn(
        `Expected ${cmd} -V to return compile details for an httpd that included SERVER_CONFIG_FILE location. Instead received: ${result}`
    );
};
const makeError = msg =>
    new Error(
        `Could not autodetect httpd.conf path: ${msg}\n\nPlease make sure your httpd is running and/or provide the path to the httpd.conf manually.`
    );
const SERVER_CONF_RE = /^\s*\-D\s+SERVER_CONFIG_FILE="([^"]+)"/im;
module.exports = async (logger = console) => {
    // get running processes
    const procs = await psl();
    // filter for httpds based on command name
    const httpdProcs = procs.filter(p => KNOWN_HTTPDS.includes(p.name));
    // map to just the command path (removing args)
    const httpdBinPaths = httpdProcs.map(p => p.cmd.split(' ')[0]);
    // filter duplicates
    const httpds = [...new Set(httpdBinPaths)];
    if (httpds.length === 0) {
        throw makeError(`no httpd-like processes were found to be running.`);
    }
    // map each to a promise for their build details that includes server config output
    const configFilePaths = httpds.map(cmd =>
        new ECRunner(cmd)
            .run('-V')
            .then(compileDetails => {
                const match = compileDetails.match(SERVER_CONF_RE);
                if (match) {
                    return match[1];
                } else {
                    warnFor(logger, cmd, compileDetails);
                }
            })
            .catch(e => {
                warnFor(logger, cmd, e);
            })
    );
    // wait for all of them to look for the positive result
    const paths = (await Promise.all(configFilePaths)).filter(p => !!p);
    // if there is more than one positive result, it's ambiguous, so error
    if (paths.length > 1) {
        throw makeError(
            `you are running multiple httpd processes! \n${JSON.stringify(
                httpdProcs,
                null,
                2
            )}`
        );
    } else if (paths.length === 0) {
        throw makeError(
            `none of these httpd-like processes produced expected output when asked! \n${JSON.stringify(
                httpdProcs,
                null,
                2
            )}`
        );
    } else {
        return paths[0];
    }
};
